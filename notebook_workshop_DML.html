2 Manual implementation of DML (in R)

2.1 Own implementation: Doing DML once

dml_once_rf <- function(dataset, idx1, idx2) {  # indices come from second function
  # Predicting treatment from the controls, using first part of the data
  rf1 <- randomForest(formula = w ~ . - y, data = dataset[idx1, ], ntree = 200)
  
  # Predicting outcome from the controls, using first part of the data
  rf2 <- randomForest(formula = y ~ . - w, data = dataset[idx1, ], ntree = 200)
  
  # Make predictions for treatment and outcome with trained model, using second half of the data
  W_hat <- predict(rf1, newdata = dataset[idx2, ])
  Y_hat <- predict(rf2, newdata = dataset[idx2, ])
  
  # Compute residuals for treatment and outcome
  W_resid <- dataset[idx2, "w"] - W_hat
  Y_resid <- dataset[idx2, "y"] - Y_hat
  
  # Regress residual of outcome on residual of treatment, obtain coefficient for treatment residual
  beta_w <- coef(lm(Y_resid ~ 0 + W_resid))
  return(beta_w)
}

2.2 Own implementation: Cross-fitting

dml_cross_rf <- function(dataset, K = 5) {
  # Generate indices for K folds
  idx_k <- vector(mode = "list", length = K)
  idx <- 1:nrow(dataset)
  for (k in 1:(K - 1)) { # draw indices for each split randomly into a list
    idx_k[[k]] <- sample(idx, 1/K * nrow(dataset))
    idx <- setdiff(idx, idx_k[[k]])}
  idx_k[[K]] <- idx

  betas_w <- numeric(K)  # Empty vector to store the coefficients 
  # Apply the dml_once() function for each fold: ML on larger, estimation on smaller sample 
  for (k in 1:K) {
    # Split sample in two parts
    idx2 <- idx_k[[k]]
    idx1 <- setdiff(1:nrow(dataset), idx2)
    
    # Run dml_once()
    betas_w[k] <- dml_once_rf(dataset, idx1, idx2)
    }
  # Average the resulting coefficients
  return(mean(betas_w))
}

import numpy as np; import pandas as pd; np.random.seed(314)

n = 1000
x = np.random.normal(size=n)
w = x + np.random.normal(size=n)      # Linear DGP
y = w + x + np.random.normal(size=n)  # Linear DGP
df_lin = pd.DataFrame({'y': y, 'w': w, 'x': x})

w = x**2 + np.random.normal(size=n)      # Nonlinear DGP
y = w + x**2 + np.random.normal(size=n)  # Nonlinear DGP
df_nonlin = pd.DataFrame({'y': y, 'w': w, 'x': x})
from sklearn.ensemble import RandomForestRegressor
from sklearn.linear_model import LinearRegression
def dml_once_rf(dataset, idx1, idx2):
  """Run DML once (for a single fold in the cross validation)."""

  # Select covariates (everything but w and y)
  covariates = dataset.drop(columns=['w', 'y'])

  # Select w and y
  w = dataset.w
  y = dataset.y

  # Predicting treatment from the controls, using first part of the data
  rf1 = (
    RandomForestRegressor(n_estimators=200)
    .fit(X=covariates.iloc[idx1, :], y=w.iloc[idx1])
  )
  
  # Predicting outcome from the controls, using first part of the data
  rf2 = (
    RandomForestRegressor(n_estimators=200)
    .fit(X=covariates.iloc[idx1, :], y=y.iloc[idx1])
  )
  
  # Make predictions for treatment and outcome with trained model, 
  # using second half of the data
  W_hat = rf1.predict(covariates.iloc[idx2, :])
  Y_hat = rf2.predict(covariates.iloc[idx2, :])
  
  # Compute residuals for treatment and outcome
  W_resid = w.iloc[idx2] - W_hat
  Y_resid = y.iloc[idx2] - Y_hat
  
  # Regress residual of outcome on residual of treatment, obtain coefficient for treatment residual
  beta_w = (
    LinearRegression(fit_intercept=False)
    .fit(X=W_resid.values.reshape(-1, 1), y=Y_resid.values.reshape(-1, 1))
    .coef_[0,0]
  )
  
  return beta_w

3.3 Own implementation: Cross-fitting

def dml_cross_rf(dataset, K = 5):
  """Run DML K times for cross validation."""

  # Get number of observations
  n_obs = dataset.shape[0]

  # Generate indices for K folds
  idx_k = []
  idx = np.arange(n_obs)
  for k in range(K - 1):  # Draw indices for each split randomly into a list
    idx_k.append(np.random.choice(idx, n_obs//K, replace=False))
    idx = np.setdiff1d(idx, idx_k[k], assume_unique=True)
  # For last fold
  idx_k.append(idx)

  betas_w = []  # Empty list to store the coefficients 
  # Apply the dml_once() function for each fold: 
  # ML on larger, estimation on smaller sample 
  for k in range(K):
    # Split sample in two parts
    idx2 = idx_k[k]
    idx1 = np.concatenate(idx_k[:k] + idx_k[k+1:])
    
    # Run dml_once()
    betas_w.append(dml_once_rf(dataset, idx1, idx2))

  # Average the resulting coefficients
  return np.mean(betas_w)

def linreg_comp(dataset):
  beta_w = (
    LinearRegression(fit_intercept=True)
    .fit(X=dataset.drop(columns='y'), y=dataset.y)
    .coef_[0] # w is first feature
  )
  return beta_w
